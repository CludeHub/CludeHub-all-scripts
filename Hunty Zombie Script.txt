--// Realistic ESP for Roblox Exploits
--// Example UI usage with your system

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera

local ESP_ENABLED = false
local SHOW_HEALTH = true
local SHOW_NAME = true
local SHOW_DISTANCE = true
local SHOW_BOX = true
local BOX_COLOR = Color3.fromRGB(255,255,255)

--// ESP Drawing Handler
local function CreateESP(player)
    if player == LocalPlayer then return end
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        local draw = {
            Box = Drawing.new("Square"),
            Name = Drawing.new("Text"),
            HealthBar = Drawing.new("Line"),
            Distance = Drawing.new("Text")
        }
        draw.Box.Thickness = 2
        draw.Box.Filled = false
        draw.Box.Color = BOX_COLOR
        draw.Box.Visible = false

        draw.Name.Size = 13
        draw.Name.Color = Color3.fromRGB(255,255,255)
        draw.Name.Center = true
        draw.Name.Outline = true
        draw.Name.Visible = false

        draw.HealthBar.Thickness = 2
        draw.HealthBar.Color = Color3.fromRGB(0,255,0)
        draw.HealthBar.Visible = false

        draw.Distance.Size = 13
        draw.Distance.Color = Color3.fromRGB(200,200,200)
        draw.Distance.Center = true
        draw.Distance.Outline = true
        draw.Distance.Visible = false

        return draw
    end
end

local espObjects = {}

--// Update ESP
RunService.RenderStepped:Connect(function()
    for _,player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local char = player.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if hrp and hum and ESP_ENABLED then
                if not espObjects[player] then
                    espObjects[player] = CreateESP(player)
                end
                local esp = espObjects[player]
                local pos, vis = Camera:WorldToViewportPoint(hrp.Position)
                if vis then
                    local scale = (1 / (pos.Z)) * 1000
                    local sizeX, sizeY = 4 * scale, 5 * scale

                    -- Box
                    if SHOW_BOX then
                        esp.Box.Size = Vector2.new(sizeX, sizeY)
                        esp.Box.Position = Vector2.new(pos.X - sizeX/2, pos.Y - sizeY/2)
                        esp.Box.Visible = true
                    else
                        esp.Box.Visible = false
                    end

                    -- Name
                    if SHOW_NAME then
                        esp.Name.Position = Vector2.new(pos.X, pos.Y - sizeY/2 - 12)
                        esp.Name.Text = player.Name
                        esp.Name.Visible = true
                    else
                        esp.Name.Visible = false
                    end

                    -- Health
                    if SHOW_HEALTH then
                        local hpPercent = hum.Health / hum.MaxHealth
                        esp.HealthBar.From = Vector2.new(pos.X - sizeX/2 - 4, pos.Y + sizeY/2)
                        esp.HealthBar.To = Vector2.new(pos.X - sizeX/2 - 4, pos.Y + sizeY/2 - sizeY * hpPercent)
                        esp.HealthBar.Visible = true
                    else
                        esp.HealthBar.Visible = false
                    end

                    -- Distance
                    if SHOW_DISTANCE then
                        local dist = math.floor((hrp.Position - Camera.CFrame.Position).Magnitude)
                        esp.Distance.Text = tostring(dist).."m"
                        esp.Distance.Position = Vector2.new(pos.X, pos.Y + sizeY/2 + 12)
                        esp.Distance.Visible = true
                    else
                        esp.Distance.Visible = false
                    end
                else
                    for _,v in pairs(esp) do
                        v.Visible = false
                    end
                end
            end
        end
    end
end)

local RunService = game:GetService("RunService")
local farming = false
local farmConnection

autofarm:AddToggle('Enable Auto Farm', false, function(val)
    farming = val
    if farming then
        farmConnection = RunService.Heartbeat:Connect(function()
            local zombies = {}
            for _, mob in pairs(game.Workspace.Entities.Zombie:GetChildren()) do
                if mob:IsA("Model") and mob:FindFirstChild("HumanoidRootPart") then
                    table.insert(zombies, mob)
                end
            end

            if #zombies > 0 and game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local target = zombies[math.random(1, #zombies)]
                game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 3, 0)
            end
            task.wait(1)
        end)
    else
        if farmConnection then
            farmConnection:Disconnect()
            farmConnection = nil
        end
    end
end)
end)
